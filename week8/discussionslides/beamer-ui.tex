\section{Poisson Rate and Window
Size}\label{poisson-rate-and-window-size}

\begin{frame}{Time-Interval Example}
\phantomsection\label{time-interval-example}
\begin{block}{Call Center Arrivals}
\phantomsection\label{call-center-arrivals}
Assume arrivals follow (N\_t \sim \Pois(\lambda t)).

If the average is 3 calls per 10 minutes, then:

{[} \lambda = 0.3 \text{ calls/minute} {]}

So expected counts scale with interval length:

\begin{itemize}
\tightlist
\item
  5 minutes: (\Ex(N\_5) = 0.3 \times 5 = 1.5)
\item
  20 minutes: (\Ex(N\_\{20\}) = 0.3 \times 20 = 6)
\item
  1 hour: (\Ex(N\_\{60\}) = 18)
\end{itemize}
\end{block}
\end{frame}

\begin{frame}{Region-Size Example}
\phantomsection\label{region-size-example}
\begin{block}{Spatial Poisson Process}
\phantomsection\label{spatial-poisson-process}
Suppose emergency events occur at rate (\rho = 2) per square mile per
day.

For area (A): {[} N\_A \sim \Pois(\rho A), \quad \Ex(N\_A) = \rho A {]}

\begin{itemize}
\tightlist
\item
  (A=1): (\Ex(N\_A)=2)
\item
  (A=3.5): (\Ex(N\_A)=7)
\item
  (A=10): (\Ex(N\_A)=20)
\end{itemize}
\end{block}
\end{frame}

\section{Continuous Piecewise
Example}\label{continuous-piecewise-example}

\begin{frame}{Define a PDF}
\phantomsection\label{define-a-pdf}
\begin{block}{Two Non-Zero Segments}
\phantomsection\label{two-non-zero-segments}
{[} f\_X(x)=

\begin{cases}
x, & 0<x\le 1,\\
2-x, & 1<x<2,\\
0, & \text{otherwise}.
\end{cases}

{]}

This is a continuous-distribution example with (f(X=x)=0) for every
single point (x).
\end{block}
\end{frame}

\begin{frame}{Derive CDF from PDF}
\phantomsection\label{derive-cdf-from-pdf}
\begin{block}{Piecewise Integration}
\phantomsection\label{piecewise-integration}
{[} F\_X(x)=

\begin{cases}
0, & x\le 0,\\
\frac{x^2}{2}, & 0<x\le 1,\\
1-\frac{(2-x)^2}{2}, & 1<x<2,\\
1, & x\ge 2.
\end{cases}

{]}
\end{block}
\end{frame}

\begin{frame}{Recover PDF from CDF}
\phantomsection\label{recover-pdf-from-cdf}
\begin{block}{Differentiate by Interval}
\phantomsection\label{differentiate-by-interval}
{[} f\_X(x)=F'\_X(x)=

\begin{cases}
x, & 0<x<1,\\
2-x, & 1<x<2,\\
0, & \text{otherwise}.
\end{cases}

{]}

At isolated points (e.g., (x=1)), the exact point value does not change
probabilities.
\end{block}
\end{frame}

\begin{frame}{Compute (\Ex(X))}
\phantomsection\label{compute-x}
\begin{block}{Mean}
\phantomsection\label{mean}
{[} \Ex(X)=\int\_0\^{}1 x\cdot x,dx + \int\_1\^{}2 x(2-x),dx = 1 {]}
\end{block}
\end{frame}

\begin{frame}{Compute (\Var(X))}
\phantomsection\label{compute-x-1}
\begin{block}{Variance}
\phantomsection\label{variance}
{[} \Ex(X\^{}2)=\int\_0\^{}1 x\^{}2\cdot x,dx + \int\_1\^{}2
x\^{}2(2-x),dx = \frac{7}{6} {]}

{[} \Var(X)=\Ex(X\textsuperscript{2)-\Ex(X)}2=\frac{7}{6}-1=\frac{1}{6}
{]}
\end{block}
\end{frame}
