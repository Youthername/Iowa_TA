\section{Standardization and Z-table}
% ==============================

\subsection{Core Setup}
\begin{frame}{Standardization Formula}
  If \(X\sim\Norm(\mu,\sigma^2)\), define
  \[
    Z=\frac{X-\mu}{\sigma}\sim\Norm(0,1).
  \]
  Then
  \[
    P(X\le x)=P\!\left(Z\le \frac{x-\mu}{\sigma}\right)=\Phi\!\left(\frac{x-\mu}{\sigma}\right).
  \]
  \vspace{2mm}
  Z-table usually reports \(\Phi(z)=P(Z\le z)\).
\end{frame}

\subsection{Tail and Interval Probabilities}
\begin{frame}{Tail and Interval Forms}
  \[
    P(X>x)=1-\Phi\!\left(\frac{x-\mu}{\sigma}\right).
  \]
  \[
    P(a<X<b)=\Phi\!\left(\frac{b-\mu}{\sigma}\right)-\Phi\!\left(\frac{a-\mu}{\sigma}\right).
  \]
  \vspace{2mm}
  Common symmetry:
  \[
    \Phi(-z)=1-\Phi(z).
  \]
\end{frame}

\subsection{Z-table Reading Rules}
\begin{frame}{Z-table Quick Rules}
  Most Z-tables report left-tail area:
  \[
    \Phi(z)=P(Z\le z).
  \]
  So we convert every probability into \(\Phi(\cdot)\) form:
  \[
    P(Z>z)=1-\Phi(z),\qquad
    P(a<Z<b)=\Phi(b)-\Phi(a).
  \]
  For negative \(z\), use symmetry:
  \[
    \Phi(-z)=1-\Phi(z).
  \]
\end{frame}

\subsection{Worked Example}
\begin{frame}{Worked Example: Probability Between Two Values}
  Let \(X\sim\Norm(20,4^2)\). Find \(P(15<X<26)\).
  \[
    z_1=\frac{15-20}{4}=-1.25,\qquad z_2=\frac{26-20}{4}=1.5.
  \]
  \[
    P(15<X<26)=\Phi(1.5)-\Phi(-1.25).
  \]
  Using Z-table:
  \[
    \Phi(1.5)\approx0.9332,\qquad \Phi(-1.25)\approx0.1056.
  \]
  \[
    P(15<X<26)\approx0.8276.
  \]
\end{frame}

\section{Percentiles}
% ==============================

\subsection{General Rule}
\begin{frame}{Percentile Formula}
  The \(p\)-th percentile \(x_p\) solves \(P(X\le x_p)=p\).

  For \(X\sim\Norm(\mu,\sigma^2)\):
  \[
    x_p=\mu+z_p\sigma,
  \]
  where \(z_p\) is the \(p\)-th percentile of \(\Norm(0,1)\).

  Common value:
  \[
    z_{0.975}\approx1.96.
  \]
\end{frame}

\subsection{Find a Percentile}
\begin{frame}{Example: Find the 97.5th Percentile}
  Suppose \(X\sim\Norm(150,12^2)\). Find \(x_{0.975}\).
  \[
    x_{0.975}=150+1.96\times12=173.52.
  \]
  So the 97.5th percentile is about
  \[
    173.5.
  \]
\end{frame}

\subsection{Solve for Parameters}
\begin{frame}{Example: Solve for \(\mu\) or \(\sigma\)}
  If \(x_{0.975}=180\) and \(\sigma=12\), then
  \[
    180=\mu+1.96(12)
    \quad\Longrightarrow\quad
    \mu=156.48.
  \]

  If \(\mu\) is known instead, solve
  \[
    \sigma=\frac{x_p-\mu}{z_p}.
  \]
\end{frame}

\subsection{Inverse Percentile Template}
\begin{frame}{Inverse Percentile Template (Must Know)}
  Start from
  \[
    x_p=\mu+z_p\sigma.
  \]
  Then solve by unknown:
  \[
    \mu=x_p-z_p\sigma
    \quad\text{or}\quad
    \sigma=\frac{x_p-\mu}{z_p}.
  \]
  Reminder:
  \[
    z_{0.975}\approx1.96,\qquad z_{0.025}\approx-1.96.
  \]
  Keep the sign of \(z_p\) consistent with the percentile \(p\).
\end{frame}

\section{Linear Combinations}
% ==============================

\subsection{Mean and Variance Rules}
\begin{frame}{Linear Combination Rules}
  Let \(Y=aX_1+bX_2+c\).
  \[
    \Ex(Y)=a\Ex(X_1)+b\Ex(X_2)+c.
  \]

  If \(X_1\ind X_2\),
  \[
    \Var(Y)=a^2\Var(X_1)+b^2\Var(X_2).
  \]

  More general form:
  \[
    \Var(aX_1+bX_2)=a^2\Var(X_1)+b^2\Var(X_2)+2ab\cov(X_1,X_2).
  \]
\end{frame}

\subsection{Worked Example}
\begin{frame}{Example: \(Y=0.8X_1+0.2X_2\)}
  Suppose \(X_1,X_2\iid\Norm(20,4^2)\), and
  \[
    Y=0.8X_1+0.2X_2.
  \]
  \[
    \Ex(Y)=0.8(20)+0.2(20)=20.
  \]
  \[
    \Var(Y)=0.8^2(16)+0.2^2(16)=10.88.
  \]
  Since a linear combination of independent normals is normal,
  \[
    Y\sim\Norm(20,10.88).
  \]
\end{frame}

\section{Linear Expression Probability}
% ==============================

\subsection{Method}
\begin{frame}{How to Handle \(P(4X>Y+1)\)-Type Questions}
  Rewrite the inequality as one normal variable.

  Define
  \[
    W=4X-Y.
  \]
  Then
  \[
    P(4X>Y+1)=P(W>1).
  \]
  If \(X,Y\) are independent normals, then \(W\) is also normal.
\end{frame}

\subsection{Four-Step Template}
\begin{frame}{Four-Step Template for Linear Expressions}
  For expressions like \(P(aX+bY>c)\):
  \begin{enumerate}
    \item Define one variable: \(W=aX+bY\), so target is \(P(W>c)\).
    \item Compute moments:
    \[
      \Ex(W)=a\Ex(X)+b\Ex(Y),
    \]
    \[
      \Var(W)=a^2\Var(X)+b^2\Var(Y)+2ab\cov(X,Y).
    \]
    \item If independent, \(\cov(X,Y)=0\). Then \(W\) is normal.
    \item Standardize and use Z-table:
    \[
      P(W>c)=1-\Phi\!\left(\frac{c-\mu_W}{\sigma_W}\right).
    \]
  \end{enumerate}
\end{frame}

\subsection{Worked Example}
\begin{frame}{Example: Compute \(P(4X>Y+1)\)}
  Let \(X\sim\Norm(2,1^2)\), \(Y\sim\Norm(5,2^2)\), independent.

  Set \(W=4X-Y\). Then
  \[
    \Ex(W)=4(2)-5=3,
    \qquad
    \Var(W)=16(1^2)+2^2=20.
  \]
  So \(W\sim\Norm(3,20)\), and
  \[
    P(4X>Y+1)=P(W>1)
    =1-\Phi\!\left(\frac{1-3}{\sqrt{20}}\right)
    =\Phi(0.447)\approx0.672.
  \]
\end{frame}

\section{Sampling Distribution of Xbar}
% ==============================

\subsection{Main Result}
\begin{frame}{Theorem for \(\bar X\)}
  If \(X_1,\dots,X_n\iid\Norm(\mu,\sigma^2)\), then
  \[
    \bar X\sim\Norm\!\left(\mu,\frac{\sigma^2}{n}\right).
  \]
  Therefore
  \[
    Z=\frac{\bar X-\mu}{\sigma/\sqrt{n}}\sim\Norm(0,1).
  \]
  Also,
  \[
    S_n=\sum_{i=1}^n X_i\sim\Norm(n\mu,n\sigma^2).
  \]
\end{frame}

\subsection{Sum-to-Mean Trick}
\begin{frame}{Sum-to-Mean Trick}
  For sums, convert to \(\bar X\) first:
  \[
    P\!\left(\sum_{i=1}^n X_i>c\right)
    =P\!\left(\bar X>\frac{c}{n}\right).
  \]
  This often makes standardization cleaner because
  \[
    \bar X\sim\Norm\!\left(\mu,\frac{\sigma^2}{n}\right).
  \]
  Use this both for probability questions and percentile questions.
\end{frame}

\subsection{Probability Example}
\begin{frame}{Example: Probability for a Sample Mean}
  Suppose \(X_i\iid\Norm(20,4^2)\), \(n=25\). Find \(P(\bar X>21)\).
  \[
    \bar X\sim\Norm\!\left(20,\frac{16}{25}\right),
    \qquad
    \SE(\bar X)=\frac{4}{5}=0.8.
  \]
  \[
    P(\bar X>21)=1-\Phi\!\left(\frac{21-20}{0.8}\right)
    =1-\Phi(1.25)
    \approx0.1056.
  \]
\end{frame}

\subsection{Percentile of Sums or Averages}
\begin{frame}{Percentiles of \(\bar X\) and \(S_n\)}
  For \(\bar X\sim\Norm(\mu,\sigma^2/n)\):
  \[
    (\bar X)_p=\mu+z_p\frac{\sigma}{\sqrt{n}}.
  \]
  For \(S_n\sim\Norm(n\mu,n\sigma^2)\):
  \[
    (S_n)_p=n\mu+z_p\sqrt{n}\,\sigma.
  \]
  Same logic: convert to \(Z\), use Z-table, then convert back.
\end{frame}

\subsection{Percentile Example for Sum}
\begin{frame}{Example: Weight Limit with 0.1\% Exceedance}
  Suppose \(X_i\iid\Norm(160,30^2)\), \(n=25\), and
  \[
    Y=\sum_{i=1}^{25}X_i\sim\Norm(4000,150^2).
  \]
  Find \(y_0\) such that \(P(Y>y_0)=0.001\).

  Equivalent:
  \[
    P(Y\le y_0)=0.999
    \Rightarrow
    \frac{y_0-4000}{150}=z_{0.999}\approx3.09.
  \]
  So
  \[
    y_0\approx 4000+3.09(150)=4463.5.
  \]
\end{frame}

\section{Optional: Delta Method}
% ==============================

\subsection{Only If Asked}
\begin{frame}{Delta Method Preview (Optional)}
  Use when we need approximation for a function of a random variable, \(u(X)\).

  If \(X\) has mean \(\mu_X\), variance \(\sigma_X^2\), and \(u\) is smooth near \(\mu_X\):
  \[
    \Ex[u(X)]\approx u(\mu_X)+\frac{u''(\mu_X)}{2}\sigma_X^2,
  \]
  \[
    \Var[u(X)]\approx \bigl(u'(\mu_X)\bigr)^2\sigma_X^2.
  \]
  This is based on Taylor expansion around \(\mu_X\).
\end{frame}

\subsection{Practice 3}
\begin{frame}{Practice 3: Linear Combination Moments}
  Let \(X_1,X_2\iid\Norm(10,3^2)\), and \(Y=2X_1-X_2+5\).
  Find \(\Ex(Y)\) and \(\Var(Y)\).
  \pause
  \[
    \Ex(Y)=2(10)-10+5=15,
    \qquad
    \Var(Y)=2^2(9)+(-1)^2(9)=45.
  \]
\end{frame}

\subsection{Practice 4}
\begin{frame}{Practice 4: Linear Expression Probability}
  If \(X\sim\Norm(1,1)\), \(Y\sim\Norm(2,4)\), independent,
  compute \(P(4X>Y+1)\).
  \pause
  \[
    W=4X-Y\sim\Norm(2,20),
  \]
  \[
    P(4X>Y+1)=P(W>1)
    =1-\Phi\!\left(\frac{1-2}{\sqrt{20}}\right)
    =\Phi(0.224)\approx0.589.
  \]
\end{frame}

\subsection{Practice 5}
\begin{frame}{Practice 5: Sampling Distribution}
  If \(X_i\iid\Norm(50,9^2)\), \(n=36\), find \(P(\bar X<52)\).
  \pause
  \[
    \SE(\bar X)=\frac{9}{\sqrt{36}}=1.5,
    \qquad
    z=\frac{52-50}{1.5}=1.333.
  \]
  \[
    P(\bar X<52)=\Phi(1.333)\approx0.9082.
  \]
\end{frame}

\subsection{Practice 6}
\begin{frame}{Practice 6: Percentile of a Sum}
  Let \(X_i\iid\Norm(40,6^2)\), \(n=16\), and \(S=\sum_{i=1}^{16}X_i\).
  Find \(s_{0.95}\), the 95th percentile of \(S\).
  \pause
  \[
    S\sim\Norm(16\times40,\;16\times6^2)=\Norm(640,24^2).
  \]
  Using \(z_{0.95}\approx1.645\):
  \[
    s_{0.95}=640+1.645(24)=679.48.
  \]
\end{frame}

\section{Wrap-Up}
% ==============================

\subsection{Take-Home Messages}
\begin{frame}{Take-Home Messages}
  \begin{itemize}
    \item Standardize first; then use the Z-table.
    \item Percentiles follow \(x_p=\mu+z_p\sigma\) (or its inverse forms).
    \item For independent variables, linear-combination variance adds by squared coefficients.
    \item Convert expressions and sample means into one normal variable before computing probability.
  \end{itemize}
\end{frame}
